29  package org.imageterrier.basictools;

31  import java.io.File;
32  import java.io.IOException;
33  import java.io.StringReader;
34  
35  import org.apache.log4j.xml.DOMConfigurator;
36  
37  import org.terrier.utility.ApplicationSetup;
38  
39  /**
40   * Utility to help enable the use of Terrier without having 
41   * to install anything
42   * 
43   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
44   *
45   */
46  public class BasicTerrierConfig {
47  	/**
48  	 * log4j configuration
49  	 */
50  	public static final String DEFAULT_LOG4J_CONFIG = 
51  		  "<?xml version=\"1.0\" encoding=\"UTF-8\" ?>"
52  		+ "<!DOCTYPE log4j:configuration SYSTEM \"log4j.dtd\">" 
53  		+ "<log4j:configuration xmlns:log4j=\"http://jakarta.apache.org/log4j/\">"
54  		+ " <appender name=\"console\" class=\"org.apache.log4j.ConsoleAppender\">"
55  		+ "  <param name=\"Target\" value=\"System.err\"/>"
56  		+ "  <layout class=\"org.apache.log4j.SimpleLayout\"/>"
57  		+ " </appender>"
58  		+ " <logger name=\"org.terrier\">"
59  		+ "  <level value=\"__LEVEL__\" />"
60  		+ "  <appender-ref ref=\"console\" />"
61  		+ " </logger>"
62  		+ " <logger name=\"org.imageterrier\">"
63  		+ "  <level value=\"__LEVEL__\" />"
64  		+ "  <appender-ref ref=\"console\" />"
65  		+ " </logger>"
66  		+ "</log4j:configuration>";
67  	
68  	/**
69  	 * The log level property string
70  	 */
71  	public static String LOG_LEVEL = "imageterrier.log.level";
72  	
73  	private static boolean isConfigured = false;
74  	
75  	/**
76  	 * This method attempts to aid the circumvention of the terrier ApplicationSetup badness!
77  	 * We should really re-write that bit of terrier...
78  	 */
79  	public static synchronized void configure() {
80  		if (isConfigured)
81  			return;
82  		
83  		try {
84  			File tmp = File.createTempFile("terrier-tmp-", ".properties");
85  			
86  			System.setProperty("terrier.home", tmp.getParent());
87  			System.setProperty("terrier.setup", tmp.getAbsolutePath());
88  
89  			ApplicationSetup.setProperty("termpipelines", "");
90  			ApplicationSetup.setProperty("querying.postprocesses.order", "");
91  			ApplicationSetup.loadCommonProperties(); //THIS IS REQUIRED TO PROPERLY READ PREFS (ApplicationSetup is a POS!!)
92  			
93  			ApplicationSetup.TERRIER_INDEX_PREFIX = "index"; //imageterrier convention
94  			
95  			//reconfigure log4j
96  			new DOMConfigurator().doConfigure(new StringReader(DEFAULT_LOG4J_CONFIG.replace("__LEVEL__", System.getProperty(LOG_LEVEL, "warn"))), org.apache.log4j.LogManager.getLoggerRepository());
97  			
98  			tmp.delete();
99  			isConfigured = true;
100 			
101 		} catch (IOException e) {
102 			e.printStackTrace(System.err);
103 			System.err.println("Error with terrier configuration");
104 			System.exit(1);
105 		}
106 	}
107 }
