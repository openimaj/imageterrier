75  	public QLFDocument<F> clone() {
76  		ArrayList<F> newFeatures = new ArrayList<F>(this.featureList.size());
77  		for (F f : this.featureList) {
78  			newFeatures.add(f);
79  		}

81  		QLFDocument<F> ret = new QLFDocument<F>(newFeatures, docno, props);

171 	/**
172 	 * Convenience methods to filter all features that have spatial locations 
173 	 * outside the given rectangle. It is assumed that the 0th ordinate of the
174 	 * feature location is the x-ordinate, and the 1st is the y ordinate.
175 	 * @param rect the rectangle.
176 	 */
177 	public void filter(Rectangle rect) {
178 		float x1 = rect.x;
179 		float y1 = rect.y;
180 		float x2 = rect.x + rect.width;
181 		float y2 = rect.y + rect.height;
182 		
183 		List<F> newEntries = new ArrayList<F>();
184 		TIntObjectHashMap<List<F>> newIndex = new TIntObjectHashMap<List<F>>();
185 	
186 		//TODO: this is currently selecting regions that overlap the bounds...
187 		//we might want to stop that...
188 		for (F le : featureList) {
189 			if (le.getLocation().getOrdinate(0).floatValue() >= x1 && le.getLocation().getOrdinate(0).floatValue() <= x2 &&
190 					le.getLocation().getOrdinate(1).floatValue() >= y1 && le.getLocation().getOrdinate(1).floatValue() <= y2) {
191 				newEntries.add(le);
192 				
193 				if (!newIndex.containsKey(le.id))
194 					newIndex.put(le.id, new ArrayList<F>());
195 				newIndex.get(le.id).add(le);
196 			}
197 		}
198 		featureList = newEntries;
199 		iterator = featureList.iterator();
200 		current = null;
201 	}

221 	/**
222 	 * Get the spatially nearest neighbouring terms to the
223 	 * target term. It is assumed that the 0th ordinate of the
224 	 * feature location is the x-ordinate, and the 1st is the y ordinate.
225 	 * @param target the target term. 
226 	 * @param nNeighbours the number of neighbours.
227 	 * @return a list of neighbouring terms
228 	 */
229 	public List<F> getNearestNeighbours(F target, int nNeighbours) {
230 		SortedSet<DistEntry> neighbourDist = new TreeSet<DistEntry>();
231 		
232 		for (F le : featureList) {
233 			if (le != target) {
234 				float x1 = le.getLocation().getOrdinate(0).floatValue();
235 				float y1 = le.getLocation().getOrdinate(1).floatValue();
236 				
237 				float x2 = target.getLocation().getOrdinate(0).floatValue();
238 				float y2 = target.getLocation().getOrdinate(1).floatValue();
239 				
240 				DistEntry de = new DistEntry();
241 				
242 				float dx = x1-x2;
243 				float dy = y1-y2;
244 				
245 				de.distance = dx*dx + dy*dy;
246 				de.le = le;
247 				
248 				neighbourDist.add(de);
249 			}
250 		}
251 		
252 		int count = 0;
253 		List<F> neighbours = new ArrayList<F>();
254 		for (DistEntry de : neighbourDist) {
255 			if (count >= nNeighbours) break;
256 			neighbours.add(de.le);
257 			
258 			count++;
259 		}
260 		
261 		return neighbours;
262 	}

272 		List<F> nns = getNearestNeighbours(current, limit);

274 		for (int i=0; i<ids.length; i++)

291 	void makeTree() {
292 		List<Pt> pts = new ArrayList<Pt>();
293 		for (F f : featureList) {
294 			pts.add(new Pt(f.getLocation().getOrdinate(0).floatValue(), f.getLocation().getOrdinate(1).floatValue(), f));
295 		}
296 		
297 		tree = new CoordinateKDTree<Pt>(pts);
298 	}

300 	/**
301 	 * Get the spatially nearest neighbouring terms to the
302 	 * target term. It is assumed that the 0th ordinate of the
303 	 * feature location is the x-ordinate, and the 1st is the y ordinate.
304 	 * 
305 	 * This method uses an underlying KD-Tree to speed the neighbour operation.
306 	 * 
307 	 * @param target the target term. 
308 	 * @param nNeighbours the number of neighbours.
309 	 * @return a list of neighbouring terms
310 	 */
311 	public int [] getNearestNeighboursKD(F target, int nNeighbours) {
312 		if (tree == null) makeTree();
313 		
314 		List<Pt> pts = new ArrayList<Pt>();
315 		tree.fastKNN(pts, new Pt(target.getLocation().getOrdinate(0).floatValue(), target.getLocation().getOrdinate(1).floatValue(), target), nNeighbours);
316 		
317 		int [] ids = new int[pts.size()];
318 		
319 		for (int i=0; i<ids.length; i++) {
320 			ids[i] = pts.get(i).f.id;
321 		}
322 		
323 		return ids;
324 	}
