43  /**
44   * Reads an InvertedIndex containing payload information as a stream
45   * 
46   * @author Jonathon Hare <jsh2@ecs.soton.ac.uk>
47   *
48   * @param <PAYLOAD> the type of payload
49   */
50  public class TermPayloadInvertedIndexInputStream<PAYLOAD> extends InvertedIndexInputStream implements IndexConfigurable {
51  	protected int DocumentBlockCountDelta = 1;
52  	protected TermPayloadCoordinator<PAYLOAD> payloadConf;
53  
54  	/**
55  	 * Construct the stream
56  	 * @param index the parent index
57  	 * @param structureName the name of the inverted index
58  	 * @param postingIteratorClass the type of posting iterator
59  	 * @throws IOException
60  	 */
61  	public TermPayloadInvertedIndexInputStream(Index index, String structureName, Class<? extends IterablePosting> postingIteratorClass) throws IOException {
62  		super(index, structureName, postingIteratorClass);
63  	}
64  
65  	/**
66  	 * Construct the stream
67  	 * @param index the parent index
68  	 * @param structureName the name of the inverted index
69  	 * @param lexInputStream input stream for the lexicon
70  	 * @param postingIteratorClass the type of posting iterator
71  	 * @throws IOException
72  	 */
73  	public TermPayloadInvertedIndexInputStream(Index index, String structureName, Iterator<? extends Pointer> lexInputStream, Class<? extends IterablePosting> postingIteratorClass) throws IOException {
74  		super(index, structureName, lexInputStream, postingIteratorClass);
75  	}
76  
77  	/** let it know which index to use */
78  	@Override
79  	public void setIndex(Index i)
80  	{
81  		DocumentBlockCountDelta = i.getIntIndexProperty("blocks.invertedindex.countdelta", 1);
82  		
83  		//read in the index properties for term Payloads
84  		payloadConf = TermPayloadCoordinator.create(i);
85  	}
86  
87  	@SuppressWarnings("unchecked")
88  	@Override
89  	protected IterablePosting loadPostingIterator(BitIndexPointer pointer) throws IOException {
90  		IterablePosting ip = super.loadPostingIterator(pointer);
91  		
92  		if (ip instanceof TermPayloadIterablePosting) {
93  			((TermPayloadIterablePosting<PAYLOAD>)ip).setPayloadConfig(payloadConf);
94  		}
95  		
96  		return ip;
97  	}
98  	
99  	@Override
100 	protected int[][] getNextDocuments(BitIndexPointer pointer) throws IOException {
101 		//System.err.println("pointer="+pointer.toString() + " actual=@{"+file.getByteOffset() + ","+ file.getBitOffset()+ "}");
102 		if (file.getByteOffset() != pointer.getOffset())
103 		{
104 			//System.err.println("skipping " + (pointer.getOffset() - file.getByteOffset()) + " bytes");
105 			file.skipBytes(pointer.getOffset() - file.getByteOffset());
106 		}
107 		if (file.getBitOffset() != pointer.getOffsetBits())
108 		{
109 			//System.err.println("skipping "+ (pointer.getOffsetBits() - file.getBitOffset()) + "bits");
110 			file.skipBits(pointer.getOffsetBits() - file.getBitOffset());
111 		}
112 		
113 		final int df = pointer.getNumberOfEntries();
114 		
115 		final int[][] documentTerms = new int[5][];
116 		documentTerms[0] = new int[df];
117 		documentTerms[1] = new int[df];
118 		documentTerms[2] = new int[df];
119 		documentTerms[3] = new int[df];
120 
121 		documentTerms[0][0] = file.readGamma() - 1; //docid
122 		documentTerms[1][0] = file.readUnary();		//freq
123 
124 		int numOfTerms = documentTerms[3][0] = documentTerms[1][0];
125 		for (int i=0; i<numOfTerms; i++) {
126 			payloadConf.readPayload(file);
127 		}
128 
129 		for (int i = 1; i < df; i++) {					
130 			documentTerms[0][i]  = file.readGamma() + documentTerms[0][i - 1];
131 			documentTerms[1][i]  = file.readUnary();
132 
133 			numOfTerms = documentTerms[3][i] = documentTerms[1][i];
134 			for (int k=0; k<numOfTerms; k++) {
135 				payloadConf.readPayload(file);
136 			}
137 		}
138 
139 		//documentTerms[4] = blockids.toNativeArray();
140 		return documentTerms;
141 	}

144 	public void print() {
145 		int documents[][] = null;
146 		int i =0;
147 		try{
148 			while((documents = getNextDocuments()) != null)
149 			{
150 				System.out.print("tid"+i);
151 				int blockindex = 0;
152 				for (int j = 0; j < documents[0].length; j++) {
153 					System.out.print(
154 							"("
155 							+ documents[0][j]
156 							               + ", "
157 							               + documents[1][j]
158 							                              + ", ");
159 					if (super.fieldCount>0)
160 					{
161 						System.out.print(documents[2][j]
162 						                              + ", ");
163 					}
164 					System.out.print( documents[3][j]);
165 
166 					for (int k = 0; k < documents[3][j]; k++) {
167 						System.out.print(", B" + documents[4][blockindex]);
168 						blockindex++;
169 					}
170 					System.out.print(")");
171 				}
172 				System.out.println();
173 			}
174 		} catch (IOException ioe) {}
175 	}
