32  import org.imageterrier.locfile.PositionSpec.PositionSpecMode;

35  public class ConsistentOriScoreModifier extends AbstractHistogramConsistentScore {
36  	@Override
37  	public String getName() {
38  		return "ConsistentOriScoreModifier";
39  	}
40  
41  	@Override
42  	public int[] getPositionSpecIndices(PositionSpec spec) {
43  		int oriidx;
44  		if (spec.getMode() == PositionSpecMode.ORI) {
45  			oriidx = 0;
46  		} else if (spec.getMode() == PositionSpecMode.SPATIAL_SCALE_ORI) {
47  			oriidx = 3;
48  		} else {
49  			return null;
50  		}
51  		return new int[]{oriidx};
52  	}
53  
54  	@Override
55  	public void incr(int[][] matchedTermPos, int[] queryTermIndecies,float[] hist, double[] maxVals) {
56  		for (int i=0; i<matchedTermPos.length; i++) {
57  			//double difTheta = (queryTermIndecies[0] - matchedTermPos[i][0] + maxVals[0]) / (2 * maxVals[0]);
58  			double difTheta = (queryTermIndecies[0] - matchedTermPos[i][0]);
59  			
60  			if (difTheta > maxVals[0]/2) difTheta -= maxVals[0];
61  			if (difTheta < -maxVals[0]/2) difTheta += maxVals[0];
62  			
63  			difTheta += maxVals[0] / 2;
64  			difTheta /= (maxVals[0] + 0.00001);
65  			
66  			difTheta *= hist.length;
67  			int bin = (int)(difTheta);
68  			double rem = difTheta - bin;
69  			
70  			int obin = bin-1; 
71  			if (rem > 0.5) {
72  				obin = bin+1;
73  			}
74  			
75  			if (obin < 0) obin = hist.length-1;
76  			if (obin >= hist.length) obin = 0;
77  			
78  			double ofrac = Math.abs(rem-0.5);
79  			double frac = 1 - ofrac; 
80  			
81  			hist[bin] += frac;
82  			hist[obin] += ofrac;
83  		}
84  	}
